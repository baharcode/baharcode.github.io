    <h1>Projelerim</h1>
    <div id="proje-listesi" class="portfolyo-kapsayici">
        </div>

    <script>
        // Verileri JSON dosyasından çekiyoruz
        fetch('projeler.json')
            .then(response => response.json())
            .then(data => {
                const liste = document.getElementById('proje-listesi');
                data.forEach(proje => {
                    liste.innerHTML += `
                        <div class="kart">
                            <img src="${proje.resim}" alt="${proje.ad}">
                            <h3>${proje.ad}</h3>
                            <p>${proje.detay}</p>
                            <a href="${proje.link}" target="_blank">İncele</a>
                        </div>
                    `;
                });
            });
    </script>
